use std::fmt;

// A custom error type for weight validation.
// This is more descriptive than just returning a string.
#[derive(Debug, PartialEq)]
pub enum WeightError {
    OutOfRange(f64),
}

impl fmt::Display for WeightError {
    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {

    pub edges: Vec<String>,
    pub created_at: chrono::NaiveDateTime,

        created_at: chrono::NaiveDateTime,

            return Err(WeightError::OutOfRange(weight));
        }

        Ok(MemoryNode {
            id,
            label,
            weight,
            edges,

    *   The `new` function returns `Err(WeightError::OutOfRange(weight))` if the validation fails, and `Ok(MemoryNode { ... })` if it succeeds.

5.  **Serialization**:

    *   To deserialize, you would use `serde_json::from_str`.
    *   The `#[derive(Serialize, Deserialize)]` macro automatically generates the necessary code for these operations.
