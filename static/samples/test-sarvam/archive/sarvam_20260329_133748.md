import re
from collections import defaultdict

def group_logs(log_lines):
    """
    Groups log messages by their level. Malformed lines are grouped under 'MALFORMED'.
    
    Args:
        log_lines (list of str): List of log lines in the format '[TIMESTAMP] [LEVEL] message'.
    
    Returns:
        dict: A dictionary where keys are log levels or 'MALFORMED', and values are lists of messages.
    """
    result = defaultdict(list)
    pattern = re.compile(r'^$$[^$$]+$$\s+([A-Z]+)\s+(.+)$')
    
    for line in log_lines:
        match = pattern.match(line)
        if match:
            level = match.group(1)
            message = match.group(2)
            result[level].append(message)
        else:
            result['MALFORMED'].append(line)
    
    return dict(result)

use regex::Regex;
use std::collections::HashMap;

fn group_logs(log_lines: Vec<String>) -> HashMap<String, Vec<String>> {
    let re = Regex::new(r"$$[^$$]+$$\s+([A-Z]+)\s+(.+)$").unwrap();
    let mut result = HashMap::new();

    for line in log_lines {
        if let Some(caps) = re.captures(&line) {
            let level = caps.get(1).unwrap().as_str().to_string();
            let message = caps.get(2).unwrap().as_str().to_string();
            result.entry(level).or_insert_with(Vec::new).push(message);
        } else {
            result.entry("MALFORMED".to_string())
                .or_insert_with(Vec::new)
                .push(line);
        }
    }

    result
}

fn main() {
    let test_data = vec![
        "[2023-01-01] INFO User logged in".to_string(),
        "[2023-01-02] ERROR Failed to connect".to_string(),
        "MALFORMED LINE".to_string(),
        "[2023-01-03] WARNING Low disk space".to_string(),
        "Another malformed line".to_string(),
        "[2023-01-04] DEBUG Debug message".to_string(),
    ];

    let result = group_logs(test_data);
    for (key, value) in &result {
        println!("{}: {:?}", key, value);
    }
}

import asyncio
import aiohttp

async def fetch_url(url: str, session: aiohttp.ClientSession, semaphore: asyncio.Semaphore) -> tuple:
    """
    Fetch a single URL with a semaphore to control concurrency.
    Returns a tuple (url, status_code or error message).
    """
    async with semaphore:
        try:
            async with session.get(url, timeout=10) as response:
                return (url, response.status)
        except (aiohttp.ClientError, asyncio.TimeoutError) as e:
            return (url, f"Error: {str(e)}")

async def fetch_urls(urls: list) -> dict:
    """
    Fetch multiple URLs concurrently with a maximum of 3 concurrent requests.
    Returns a dictionary mapping URLs to their status codes or error messages.
    """
    semaphore = asyncio.Semaphore(3)
    async with aiohttp.ClientSession() as session:
        tasks = [asyncio.create_task(fetch_url(url, session, semaphore)) for url in urls]
        results = await asyncio.gather(*tasks)
        return dict(results)

if __name__ == '__main__':
    # Test URLs from httpbin.org
    urls = [
        'http://httpbin.org/status/200',
        'http://httpbin.org/status/404',
        'http://httpbin.org/delay/5',
        'http://httpbin.org/get',
        'http://httpbin.org/status/200',
        'http://httpbin.org/status/404',
        'http://httpbin.org/delay/5',
        'http://httpbin.org/get',
        'http://httpbin.org/status/200',
        'http://httpbin.org/status/404',
    ]

    # Run the async function and print results
    results = asyncio.run(fetch_urls(urls))
    for url, status in results.items():
        print(f"{url}: {status}")
